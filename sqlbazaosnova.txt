  id uuid NOT NULL DEFAULT gen_random_uuid(),
  plan_id uuid NOT NULL,
  activity text NOT NULL,
  assigned_to uuid,

  notes text,
  created_at timestamp with time zone DEFAULT now(),

);

  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "Users can insert their own sound usage" ON public.bird_sound_usage
  FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Supervisors can view all sound usage" ON public.bird_sound_usage
  FOR SELECT USING (
    auth.uid() IN (
      SELECT id FROM public.profiles WHERE role IN ('supervisor', 'admin')
